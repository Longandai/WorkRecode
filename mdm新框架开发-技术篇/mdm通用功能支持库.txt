--前台手电筒的校验功能
--保持字段统一的情况下可以手动扩展
create or replace view view_fnd_value as
select VALUE_ID,
VALUE_SET_ID,
VALUE,
VALUE_MEANING,
END_ACTIVE_DATE,
START_ACTIVE_DATE,
PARENT_VALUE_LOW,
PARENT_VALUE_HIGH,
SORT_ORDER,
DESCRIPTION,
ACTIVE_FLAG,
CODELIST_FLAG,
CREATED_BY,
CREATED,
LAST_UPD_BY,
LAST_UPD,
MODIFICATION_NUM,
DELETE_FLAG,
REMARK,
VALUE_MEANING_EN
from hm_fnd_values;
